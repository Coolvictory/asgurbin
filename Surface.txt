# Define and assign the current case name extracted from "DATA READER" with "Active Case Name" as a key.
# Remove "Case " from the obtained value and store it in $currentCaseName.

!my $currentCaseName= (getValue("DATA READER","Active Case Name")=~s/Case //gr);

# Extract domain names associated with the current case from the children of "Case $currentCaseName" with "DOMAIN" as a key.
# Split the obtained string by comma (,) and store it as an array in @DomainNames_list.

!my @DomainNames_list = split(/,/, (getChildren("Case $currentCaseName", "DOMAIN")=~ s/DOMAIN://gr));

# Extract components to initialize for each domain name obtained in @DomainNames_list.
# Remove "Component " from each obtained value and store it in @Component_Row_list.

!my @Component_Row_list = map {s/Component //gr } map {getValue($_, "Components to Initialize") } @DomainNames_list;

# Create a list of domain names and associated component rows sorted by component row values.
# Each element of the list is an array reference containing domain name and associated component row.
!my @Domain_Names_Row_Dict = sort {$a->[1] <=> $b->[1]} map{[$DomainNames_list[$_], $Component_Row_list[$_]]} 0 .. $#DomainNames_list;

# Define the location values for inlet and outlet
!my $Location_Inlet = 0.03;
!my $Location_Outlet = 0.86;

# Create an array of plane names by mapping each domain name to "Inlet" and "Outlet" suffixes
!my @Plane_List = map { ( $_->[0]." Inlet", $_->[0]." Outlet" ) }@Domain_Names_Row_Dict;

!my @Streamwise_Location_List =();

!if(!@Streamwise_Location_List) {
	# Create an array of streamwise locations by mapping each domain name's position to adjusted inlet and outlet positions
	!@Streamwise_Location_List = map { ($_->[1] - 1 + $Location_Inlet, $_->[1] - 1 + $Location_Outlet)}@Domain_Names_Row_Dict;
!}

!my $Workingdictory = (split/\.res/, getValue("DATA READER","Current Results File"), 2)[0];

# Declaration of empty hash %values and variableMap
!my %values;
!my %variableMap = (
 !'N' => ["Pressure", "Total Pressure", "Total Pressure", "Total Temperature", "Temperature","Total Enthalpy", "Static Enthalpy", "Specific Volume", "Velocity", "Velocity","Velocity Meridional", "Velocity Circumferential", "Mach Number", "Mach Number", "VelocityFlow Angle", "Velocity in Stn Frame Flow Angle", "Specific Heat Capacity at Constant Pressure", "Specific Heat Capacity at Constant Volume", "sl.Conservative Mass Fraction", "Rotation Velocity"],
 !'R' => ["Pressure", "Total Pressure in Stn Frame", "Total Pressure in Rel Frame","Total Temperature in Stn Frame", "Temperature", "Total Enthalpy in Stn Frame","Static Enthalpy", "Specific Volume", "Velocity in Stn Frame", "Velocity", "Velocity Meridional", "Velocity in Stn Frame Circumferential","Mach Number in Stn Frame", "Mach Number", "Velocity Flow Angle", "Velocity in Stn Frame Flow Angle", "Specific Heat Capacity at Constant Pressure", "Specific Heat Capacity at Constant Volume", "sl.Conservative Mass Fraction", "Rotation Velocity"]
!)

# Define the file name including the directory path
!my $filepath = $Workingdictory."/output.csv";

# Open the file for writing, or die if unsuccessful
!open(my $fh, '>', $filepath) or die "Cannot open file: $!"; {

	!print $fh "Prameters"; # Writing header for parameters in CSV
	
	!foreach my $plane(@Plane_List){ # Looping through plane list
		!print $fh ",$plane"; # Writing plane names in CSV 
	!}
	
	!print $fh "\n"; # Moving to new line after writing header and plane names

	!for(my $i = 0; $i<scalar(@Plane_List); $i++){ # Looping through plane list indices
		!my $plane = $Plane_List[$i]; # Getting the current plane
		
		# Retrieving appropriate variable array based on the plane type
		!$variableArray = $plane =~ /^R/? $variableMap{'R'} :
		$variableMap{'N'} || die "Invalid plane type: $plane";
		
		# Extracting Domain from the plane
		!my $Domain = (split / /, $plane)[0];

		# Processing various data for current plane and storing them in arrays
		!my @current_values;
		
		# Angular Velocity
		# Extract angular velocities from the specified domain and split the string at '[' character, taking the first part or defaulting to "0" if not found
		# Push extracted angular velocities into the current values array
		
		!my @Angular_Vel = (split /\[/, getValue($Domain, "Angular Velocity"),2)[0]|| "0" ;
		!push @current_values, @Angular_Vel;	
		
		#No of Passages in 360
		!my @No_Passages = getValue($Domain ,"Number of Passages in 360");
		!push @current_values, @No_Passages;
		
		#Torque
		# Calling the torque function with the domain concatenated with " Blade" to specify the blade, and passing the principal axis value for the domain.
		!my @Torque = torque($Domain." Blade", getValue($Domain,"Principal Axis"));
		!push @current_values, @Torque;

		#Area
		!my @Area = area("$plane");
		!push @current_values, @Area;
		
		#Mass Flow
		!my @massFlow = massFlow("$plane");
		!push @current_values, @massFlow;
		
		#Hub Raidus
		!my @HubRaidus = minVal("Radius","$plane");
		!push @current_values, @HubRaidus;
		
		#Tip Raidus
		!my @TipRaidus = maxVal("Radius","$plane");
		!push @current_values, @TipRaidus;
		
		#Mach Number
		!my @MachNumber = maxVal("Mach Number","$plane");
		!push @current_values, @MachNumber;
		
		#Density
		!my @Density = maxVal("Density", "$plane");
		!push @current_values, @Density;
		
		# Iterating over each variable and calculating its value for the current plane
		!foreach my $variable (@$variableArray) {
		!my @variable_value = massFlowAve("$variable", "$plane");
		
		!push @current_values, @variable_value;
		
		!}

	# Define an array containing parameter names
	!my @paramater_names = ( "Angular Velocity", "No of Passages in 360", "Torque", "Area", "Mass Flow", "Hub Raidus" , "Tip Radius", "Mach Number", "Density", "Pressure", "Total Pressure", "Total Pressure in Rel Frame", "Total Temperature", "Temperature","Total Enthalpy", "Static Enthalpy", "Specific Volume", "Absolute Velocity", "Relative Velocity", "Meridional Velocity", "Cercumferential Velocity", "Absolute Mach No", "Relative Mach No", "Flow Angle" , "Swirl Angle", "Cp", "Cv", "Wetness", "Rotation Velocity");
	
	# Storing the processed values into the hash %values
	!foreach my $index (0..scalar(@paramater_names)) {
		!push @{$values{"$paramater_names[$index]"}}, $current_values[$index];
		!}
	!}


	# Defining the new order for parameter names
	!my @Paramater_Names_Neworder = ("Angular Velocity", "No of Passages in 360", "Area", "Hub Raidus", "Tip Radius", "Rotation Velocity", "Torque", "Pressure", "Total Pressure", "Total Pressure in Rel Frame", "Total Temperature", "Temperature","Total Enthalpy", "Static Enthalpy", "Wetness", "Mass Flow", "Absolute Velocity", "Relative Velocity", "Meridional Velocity", "Cercumferential Velocity", "Absolute Mach No", "Relative Mach No", "Swirl Angle", "Flow Angle", "Mach Number", "Density", "Specific Volume", "Cp", "Cv");

	# Writing parameter names and their values in the new order to the CSV file
	!foreach my $Paramater_Names_Neworder (@Paramater_Names_Neworder)
	!{
		!print $fh "$Paramater_Names_Neworder";
		!foreach my $value (@{$values{$Paramater_Names_Neworder}})
		!{
			!print $fh ",$value";
		!}
		!print $fh "\n";
	!}


	# Printing the current case name to the file handle $fh
	!print $fh "$currentCaseName\n";

	# Calculating the number of Stage by excluding diffuser domain
	# If a domain starts with 'D', it's excluded from the count
	!my $No_of_Domain = (grep { $_ =~ /^D/ } @DomainNames_list) ? (scalar @DomainNames_list - 1)/2 : (scalar @DomainNames_list)/2;

	# Printing the calculated number of Stage to the file handle $fh
	!print $fh "$No_of_Domain\n";

	# Adding an empty line for readability in the file
	!print $fh "\n";
!}
!close($fh) or die "Cannot close file:$!";
